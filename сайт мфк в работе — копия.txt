        let heroHeight = 0;
        let documentIndex = 0;
        
        let rotationAngle = 0;
        let parallaxImages = document.querySelectorAll('.parallax-section-image');

        function rotateImages() {
            rotationAngle += 0.1; // Скорость вращения
            parallaxImages.forEach(img => {
                const initialRotate = parseFloat(img.dataset.initialRotate);
                img.style.transform = `rotate(${initialRotate + rotationAngle}deg)`;
            });
            requestAnimationFrame(rotateImages);
        }

        const heroSection = document.getElementById('home');
        const heroPngs = document.querySelectorAll('.hero-png');
        const pngComposition = document.getElementById('png-composition');

        const flyingPngs = document.querySelectorAll('.parallax-flying-image');
        
        let mouseOffsetX = 0;
        let mouseOffsetY = 0;

        // Обновленная функция для параллакса от мыши
        function updateHeroPngsWithMouse() {
            heroPngs.forEach(png => {
                const originalX = parseFloat(png.dataset.originalX);
                const originalY = parseFloat(png.dataset.originalY);
                const originalRotate = parseFloat(png.dataset.originalRotate);
                const depth = parseFloat(png.dataset.depth) || 0; // Глубина эффекта для каждого изображения

                // Расчет смещения на основе позиции мыши и глубины
                // Изображения на переднем плане (больший depth) будут двигаться сильнее
                const mouseTranslateX = -mouseOffsetX * 50 * depth;
                const mouseTranslateY = -mouseOffsetY * 50 * depth;
                
                // Комбинируем начальную позицию с эффектом от мыши
                const combinedTranslateX = originalX + mouseTranslateX;
                const combinedTranslateY = originalY + mouseTranslateY;

                png.style.transform = `translate(${combinedTranslateX}px, ${combinedTranslateY}px) rotate(${originalRotate}deg)`;
            });
        }
        
        // Добавление data-атрибутов для исходных позиций PNG изображений главной страницы
        document.addEventListener('DOMContentLoaded', () => {
            heroHeight = heroSection.offsetHeight;
        
            heroPngs.forEach(png => {
                const style = window.getComputedStyle(png);
                const transformMatrix = new DOMMatrixReadOnly(style.transform);
        
                const currentTranslateX = transformMatrix.m41;
                const currentTranslateY = transformMatrix.m42;
                const rotation = Math.atan2(transformMatrix.m21, transformMatrix.m11) * (180 / Math.PI);
        
                // Сохраняем начальные значения, чтобы к ним можно было применять смещение
                png.dataset.originalX = currentTranslateX;
                png.dataset.originalY = currentTranslateY;
                png.dataset.originalRotate = rotation;
            });
        
            // Инициализация начальных поворотов для параллакс-изображений под разделами
            const sectionParallaxImages = document.querySelectorAll('.parallax-section-image');
            sectionParallaxImages.forEach(img => {
                const style = window.getComputedStyle(img);
                const transformMatrix = new DOMMatrixReadOnly(style.transform);
                const rotation = Math.atan2(transformMatrix.m21, transformMatrix.m11) * (180 / Math.PI);
                img.dataset.initialRotate = `${rotation}`; 
            });

            // Инициализация летающих PNG
            flyingPngs.forEach(png => {
                const initialTop = parseFloat(png.dataset.initial-top);
                const initialLeft = parseFloat(png.dataset.initial-left);
                const initialRotate = parseFloat(png.dataset.initial-rotate);
                png.style.top = `${initialTop}%`;
                png.style.transform = `translateX(${initialLeft}px) rotate(${initialRotate}deg)`;
            });

            rotateImages();
        });


        // Обработчик события прокрутки
        window.addEventListener('scroll', () => {
            const scrollY = window.scrollY;
            const scrollFactor = scrollY / heroHeight;

            // Логика для смены цвета фона
            const body = document.body;
            const teamSection = document.getElementById('team');
            const contactsSection = document.getElementById('contacts');
            const blueSectionStart = teamSection.offsetTop;
            const blueSectionEnd = contactsSection.offsetTop + contactsSection.offsetHeight;

            if (scrollY > heroHeight / 2) {
                body.classList.add('bg-custom-blue');
                body.classList.remove('bg-gray-100');
            } else {
                body.classList.remove('bg-custom-blue');
                body.classList.add('bg-gray-100');
            }

            // Логика для летающих PNG-изображений
            if (scrollY > blueSectionStart - window.innerHeight && scrollY < blueSectionEnd) {
                const scrollableHeight = blueSectionEnd - (blueSectionStart - window.innerHeight);
                const progress = (scrollY - (blueSectionStart - window.innerHeight)) / scrollableHeight;
                
                flyingPngs.forEach((png) => {
                    // Устанавливаем видимость, когда изображения находятся в пределах синей секции
                    png.style.opacity = '0.8';

                    const direction = parseFloat(png.dataset.direction);
                    const initialLeft = (direction === 1) ? -png.offsetWidth : window.innerWidth;
                    const initialRotate = parseFloat(png.dataset.initialRotate);
                    
                    let startX = (direction === 1) ? -png.offsetWidth : window.innerWidth;
                    let endX = (direction === 1) ? window.innerWidth : -png.offsetWidth;

                    const totalDistance = Math.abs(endX - startX) + png.offsetWidth;
                    let currentTranslateX = startX + (progress * (endX - startX));
                    
                    if (direction === -1) {
                         currentTranslateX = window.innerWidth - (progress * (window.innerWidth + png.offsetWidth));
                    } else {
                        currentTranslateX = -png.offsetWidth + (progress * (window.innerWidth + png.offsetWidth));
                    }


                    png.style.transform = `translateX(${currentTranslateX}px) rotate(${initialRotate}deg)`;
                });
            } else {
                // Скрываем изображения, когда они не в синей секции
                flyingPngs.forEach((png) => {
                    png.style.opacity = '0';
                    const direction = parseFloat(png.dataset.direction);
                    const initialLeft = (direction === 1) ? -png.offsetWidth : window.innerWidth;
                    const initialRotate = parseFloat(png.dataset.initialRotate);
                    png.style.transform = `translateX(${initialLeft}px) rotate(${initialRotate}deg)`;
                });
            }
        });

        // Новый обработчик для параллакса от мыши
        pngComposition.addEventListener('mousemove', (e) => {
            const rect = pngComposition.getBoundingClientRect();
            const centerX = rect.left + rect.width / 2;
            const centerY = rect.top + rect.height / 2;
            
            // Нормализуем координаты мыши от -1 до 1
            mouseOffsetX = (e.clientX - centerX) / (rect.width / 2);
            mouseOffsetY = (e.clientY - centerY) / (rect.height / 2);
            
            // Вызываем функцию для обновления позиций PNG
            updateHeroPngsWithMouse();
        });
        
        // Код для галереи
        const galleryContainer = document.getElementById('gallery-container');
        const images = galleryContainer.querySelectorAll('.gallery-image');
        const totalImages = images.length;
        let currentImageIndex = 0;

        function scrollGallery(direction) {
            currentImageIndex += direction;
            if (currentImageIndex < 0) {
                currentImageIndex = totalImages - 1;
            } else if (currentImageIndex >= totalImages) {
                currentImageIndex = 0;
            }
            galleryContainer.scrollTo({
                left: currentImageIndex * galleryContainer.offsetWidth,
                behavior: 'smooth'
            });
        }
